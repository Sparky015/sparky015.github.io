35 inline Vec4 Bitmap::GetPixel(uint32 x, uint32 y) const
36 {
37 if (x >= m_Width)
38 {
39 AE_WARN("Trying to get pixel data with out of bounds x value: " << x << "(Image width is " << m_Width << ")")
40 x = m_Width - 1;
41 }
42 if (y >= m_Height)
43 {
44 AE_WARN("Trying to get pixel data with out of bounds y value: " << y << "(Image height is " << m_Height << ")")
45 y = m_Height - 1;
46 }
47
48 uint32 yOffset = y * m_Width;
49 uint32 componentOffset = (yOffset + x) * m_BytesPerTexel; // Considering the component size in the offset for the correct index
50 uint32 pixelIndex = componentOffset;
51 Vec4 component{0};
52 memcpy(&component.x, &m_MapData[pixelIndex], sizeof(T));
53 if (m_NumComponents > 1) { memcpy(&component.y, &m_MapData[pixelIndex + (1 * sizeof(T))], sizeof(T)); }
54 if (m_NumComponents > 2) { memcpy(&component.z, &m_MapData[pixelIndex + (2 * sizeof(T))], sizeof(T)); }
55 if (m_NumComponents > 3) { memcpy(&component.w, &m_MapData[pixelIndex + (3 * sizeof(T))], sizeof(T)); }
56 return component;
57 }

85 inline void ConvertEquirectangularToCubemap(const Bitmap& equirectangularImageData, std::vector<Bitmap>& outCubemap)
86 {
87 uint32 faceSideLength = equirectangularImageData.GetPixelWidth() / 4;
88
89 const uint32 numberOfFaces = 6;
90 outCubemap.reserve(numberOfFaces);
91
92 for (int i = 0; i < numberOfFaces; i++)
93 {
94 outCubemap.emplace_back(nullptr, ImageFormat::R16G16B16A16_UNORM, faceSideLength, faceSideLength);
95 }
96
97 uint32 maxPixelWidthIndex = equirectangularImageData.GetPixelWidth() - 1;
98 uint32 maxPixelHeightIndex = equirectangularImageData.GetPixelHeight() - 1;
99
100 for (int faceIndex = 0; faceIndex < numberOfFaces; faceIndex++)
101 {
102 for (int x = 0; x < faceSideLength; x++)
103 {
104 for (int y = 0; y < faceSideLength; y++)
105 {
106 // Convert pixel coordinates to cartesian
107 Vec3 cartesianCoords = ConvertFaceCoordinatesToCartesian(x, y, faceIndex, faceSideLength);
108
109 // Convert cartesian to spherical
110 float rho = sqrtf(cartesianCoords.x * cartesianCoords.x + cartesianCoords.y * cartesianCoords.y);
111 float phi = atan2f(cartesianCoords.y, cartesianCoords.x);
112 float theta = atan2f(cartesianCoords.z, rho);
113
114 // Convert Spherical to UVs
115 float uClamped = (float)(phi + std::numbers::pi) / (2.0f * std::numbers::pi);
116 float vClamped = (float)((std::numbers::pi / 2.0f - theta) / std::numbers::pi);
117
118 float uScaled = uClamped * equirectangularImageData.GetPixelWidth();
119 float vScaled = vClamped * equirectangularImageData.GetPixelHeight();
120
121 int32 U = (int32)(roundf(uScaled));
122 int32 V = (int32)(roundf(vScaled));
123
124 // Clamp
125 if (U < 0) { U = 0; }
126 if (V < 0) { V = 0; }
127 if (U > maxPixelWidthIndex) { U = maxPixelWidthIndex; }
128 if (V > maxPixelHeightIndex) { V = maxPixelHeightIndex; }
129
130 const Vec4& pixelData = equirectangularImageData.GetPixel<float>(U, V);
131 outCubemap[faceIndex].SetPixel<half_float::half>(x, y, pixelData);
132 }
133 }
134 }
135 }

138 inline Vec3 ConvertFaceCoordinatesToCartesian(uint32 x, uint32 y, uint32 faceIndex, uint32 faceSideLength)
139 {
140 const float U = 2.0f * x / faceSideLength;
141 const float V = 2.0f * y / faceSideLength;
142
143 Vec3 cartesianCoords;
144 switch (faceIndex)
145 {
146 case 0: cartesianCoords = { U - 1.0f, 1.0f, V - 1.0f }; break; // Positive X
147 case 1: cartesianCoords = {1.0f - U, -1.0f, V - 1.0f }; break; // Negative X
148 case 2: cartesianCoords = { 1.0f - V, U - 1.0f, -1.0f }; break; // Positive Y (I flipped Y here for Vulkan!!)
149 case 3: cartesianCoords = { V - 1.0f, U - 1.0f, 1.0f }; break; // Negative Y
150 case 4: cartesianCoords = { -1.0f, U - 1.0f, V - 1.0f }; break; // Positive Z
151 case 5: cartesianCoords = { 1.0f, 1.0f - U, V - 1.0f }; break; // Negative Z
152 default: AE_ERROR("Unsupported face index!");
153 }
154 return cartesianCoords;
155 }
